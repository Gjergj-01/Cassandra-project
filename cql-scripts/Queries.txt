# Query 1

SELECT *
FROM movies.ratings
WHERE rating = 'TV-MA' AND type = 'Movie'
LIMIT 10; 



# Query 2

SELECT show_id, title
FROM movies.netflix_titles
WHERE type = 'TV Show' AND rating = 'TV-14'
LIMIT 10;

FROM movies.netflix_titles
WHERE type = 'TV Show' AND rating = 'TV-14' 
LIMIT 10

# Query 3

# Query 4

FROM movies.ratings 
WHERE rating = 'TV-MA'
